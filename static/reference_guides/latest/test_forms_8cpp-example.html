#include <xtd/drawing/brushes>
#include <xtd/forms/user_control>
#include <xtd/math>
class waiting_spinner_control : public xtd::forms::user_control {
public:
waiting_spinner_control() {
update_timer();
timer_.tick += {*this, &waiting_spinner_control::rotate};
}
public:
int inner_radius() const noexcept {return inner_radius_;}
waiting_spinner_control& inner_radius(int value);
int line_length() const noexcept {return line_length_;}
waiting_spinner_control& line_length(int value);
int line_width() const noexcept {return line_width_;}
waiting_spinner_control& line_width(int value);
float minimum_trail_opacity() const noexcept {return minimum_trail_opacity_;}
waiting_spinner_control& minimum_trail_opacity(float value);
int number_of_lines() const noexcept {return number_of_lines_;}
waiting_spinner_control& number_of_lines(int value);
float revolutions_pers_second() const noexcept {return revolutions_per_second_;}
waiting_spinner_control& revolutions_pers_second(float value);
float roundness() const noexcept {return roundness_;}
waiting_spinner_control& roundness(float value);
bool running() const {return running_;}
waiting_spinner_control& running(bool value) {
if (running_ == value) return *this;
current_counter_ = 0;
running_ = value;
timer_.enabled(value);
return *this;
}
float trail_fade_percentage() const noexcept {return trail_fade_percentage_;}
waiting_spinner_control& trail_fade_percentage(float value);
void start() {running(true);}
void stop() {running(false);}
protected:
xtd::drawing::size default_size() const noexcept override {return {256, 256};}
void rotate() {
}
void on_paint(xtd::forms::paint_event_args& e) override {
xtd::forms::user_control::on_paint(e);
e.graphics().fill_rectangle(xtd::drawing::brushes::transparent(), e.clip_rectangle());
e.graphics().smoothing_mode(xtd::drawing::drawing_2d::smoothing_mode::anti_alias);
e.graphics().translate_transform(e.clip_rectangle().center().x(), e.clip_rectangle().center().y());
if (current_counter_ >= number_of_lines_) current_counter_ = 0;
e.graphics().rotate_transform(360.0f / static_cast<float>(number_of_lines_));
for (int i = 0; i < number_of_lines_; ++i) {
auto rotate_angle = 360.0f * i / static_cast<float>(number_of_lines_);
//e.graphics().translate_transform(inner_radius_ + line_length_, inner_radius_ + line_length_);
//e.graphics().translate_transform(inner_radius_, 0);
auto distance = line_count_distance_from_primary(i, current_counter_, number_of_lines_);
xtd::drawing::color color = current_line_color(distance, number_of_lines_, trail_fade_percentage_, minimum_trail_opacity_, fore_color());
e.graphics().fill_rounded_rectangle(xtd::drawing::solid_brush {color}, xtd::drawing::rectangle(0, -line_width_ / 2, line_length_, line_width_), static_cast<int>(line_width_ * roundness_ / 100.0f));
e.graphics().rotate_transform(i * rotate_angle);
}
}
private:
static int line_count_distance_from_primary(int current, int primary, int total_nbr_of_lines) {
auto distance = primary - current;
if (distance < 0) distance += total_nbr_of_lines;
return distance;
}
static xtd::drawing::color current_line_color(int count_distance, int total_nbr_of_lines, float trail_fade_perc, float min_opacity, xtd::drawing::color color) {
if (count_distance == 0) return color;
const auto min_alpha = min_opacity / 100.0f;
int distance_threshold = static_cast<int>(xtd::math::ceiling((total_nbr_of_lines - 1) * trail_fade_perc / 100.0f));
if (count_distance > distance_threshold) color = xtd::drawing::color::from_argb(static_cast<int>(min_alpha * 255), color);
else {
auto alpha_diff = color.a() / 255.0f - min_alpha;
auto gradient = alpha_diff / static_cast<float>(distance_threshold + 1.0f);
auto result_alpha = color.a() / 255.0f - gradient * count_distance;
// If alpha is out of bounds, clip it.
result_alpha = xtd::math::min(1.0f, xtd::math::max(0.0f, result_alpha));
color = xtd::drawing::color::from_argb(static_cast<int>(result_alpha * 255), color);
}
return color;
}
void update_size() {
auto size = (inner_radius_ + line_length_) * 2;
this->size({size, size});
}
void update_timer() {timer_.interval_milliseconds(1000 / (number_of_lines_ * revolutions_per_second_));}
private:
int = 0;
int inner_radius_ = 10;
int line_length_ = 10;
int line_width_ = 2;
float minimum_trail_opacity_ = 3.14159265358979323846f;
int number_of_lines_ = 20;
float revolutions_per_second_ = 1.57079632679489661923f;
int roundness_ = 100.0f; // 0..100
bool running_ = false;
xtd::forms::timer timer_;
float trail_fade_percentage_ = 80.0f;
};
