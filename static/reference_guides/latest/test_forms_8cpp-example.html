#include <xtd/xtd.forms>
class my_form : public xtd::forms::form {
public:
my_form() {
text("Progress Bar, Slider, and LCD Example");
size({400, 200});
progress_bar_.parent(*this);
progress_bar_.location({10, 10});
progress_bar_.size({200, 20});
slider_.parent(*this);
slider_.location({10, 50});
slider_.size({200, 20});
slider_.minimum(0);
slider_.maximum(200);
slider_.value(100);
slider_.tick_frequency(10);
slider_.tick_style(xtd::forms::tick_style::both);
slider_.value_changed += [&] {
progress_bar_.minimum(slider_.minimum());
progress_bar_.maximum(slider_.value());
};
lcd_.parent(*this);
lcd_.location({10, 90});
lcd_.size({200, 40});
timer_.interval_milliseconds(100);
timer_.tick += [&] {
if (progress_bar_.value() < progress_bar_.maximum()) progress_bar_.value(progress_bar_.value() + 1);
else progress_bar_.value(progress_bar_.minimum());
lcd_.text(xtd::ustring::format("{}", progress_bar_.value()));
};
timer_.start();
}
private:
xtd::forms::progress_bar progress_bar_;
xtd::forms::track_bar slider_;
xtd::forms::lcd_label lcd_;
xtd::forms::timer timer_;
};
int main() {
xtd::forms::application::run(my_form());
}
